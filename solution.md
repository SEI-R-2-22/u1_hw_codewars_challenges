## Sum of Positive

function positiveSum(arr) {
let sum = 0
for (let i = 0; i < arr.length; i++) {
if (arr[i] > 0) sum += arr[i]
}
return sum
}

## Sum Arrays

function sum(numbers) {
let addSum = 0
for (let i = 0; i < numbers.length; i++) addSum += numbers[i]

}
